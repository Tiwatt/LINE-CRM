        const scriptURL = "https://script.google.com/macros/s/AKfycbx2IjP6KqcCRi2R9_RyJnXWObLCifXtnuPix6g998J8SboG_CTclkU7jm9p2QhFIP7sQg/exec";


        async function submitData() {

            const date = document.getElementById("date").value;

            const channel = document.getElementById("channel").value;

            const shop = document.getElementById("shop").value;

            const orders = document.getElementById("orders").value;

            const cost = document.getElementById("cost").value;


            if (!date || !shop || !orders || !cost) {

                alert("Please fill all fields");

                return;

            }


            const data = { date, channel, shop, orders, cost };


            try {

                const response = await fetch(scriptURL, {

                    method: "POST",

                    body: JSON.stringify(data),

                    headers: { "Content-Type": "application/json" }

                });


                if (response.ok) {

                    alert("Data saved successfully!");

                    loadData();  // โหลดข้อมูลใหม่หลังจากบันทึก

                } else {

                    alert("Error saving data!");

                }

            } catch (error) {

                console.error("Error:", error);

                alert("Failed to connect to the server.");

            }

        }


        async function loadData() {

            try {

                const response = await fetch(scriptURL);

                const data = await response.json();


                const tableBody = document.getElementById("data-table");

                tableBody.innerHTML = ""; // เคลียร์ข้อมูลเก่าออกก่อน


                data.forEach(row => {

                    const newRow = `<tr>

                        <td>${row.date}</td>

                        <td>${row.channel}</td>

                        <td>${row.shop}</td>

                        <td>${row.orders}</td>

                        <td>${row.cost}</td>

                    </tr>`;

                    tableBody.innerHTML += newRow;

                });


            } catch (error) {

                console.error("Error loading data:", error);

            }

        }


        // โหลดข้อมูลเมื่อเปิดหน้าเว็บ

        loadData();
